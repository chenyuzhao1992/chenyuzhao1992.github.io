Recusive

public int uniquePathsWithObstacles(int[][] map) {
    return helper(map, map.length - 1, map[0].length - 1);
}
private int helper(int[][] map, int x, int y) {
    if (x == 0) {
      for (int i = 0; i < map[0].length; i++) {
        if (map[0][i] == 1) {
          return 0;
        }
      }
      return 1;
    }
    if (y == 0) {
      for (int i = 0; i < map[0].length; i++) {
        if (map[i][0] == 1) {
          return 0;
        }
      }
      return 1;
    }
    if (map[x][y] == 1) {
      return 0;
    }
    return helper(map, x - 1, y) + helper(map, x, y - 1);
}

Memoziation

public int uniquePathsWithObstacles(int[][] map) {
    if (map == null || map.length == 0 || map[0].length == 0) {
      return 0;
    }
    int m = map.length, n = map[0].length;
    int[][] memo = new int[m][n];
    int i;
    for (i = 0; i < m; i++) {
      if (map[i][0] == 1) {
        break;
      }
    }
    if (i == m) {
      for (i = 0; i < m; i++) {
        memo[i][0] = 1;
      }
    }
    for (i = 0; i < n; i++) {
      if (map[0][i] == 1) {
        break;
      }
    }
    if (i == n) {
      for (i = 0; i < n; i++) {
        memo[0][i] = 1;
      }
    }
    return helper(map, m - 1, n - 1, memo);
}
private int helper(int[][] map, int x, int y, int[][] memo) {
    if (x == 0 || y == 0 || memo[x][y] > 0) {
      return memo[x][y];
    }
    if (map[x][y] == 1) {
      return 0;
    }
    memo[x][y] = helper(map, x - 1, y, memo) + helper(map, x, y - 1, memo);
    return memo[x][y];
}

Bottom Up

public int uniquePathsWithObstacles(int[][] map) {
    if (map == null || map.length == 0 || map[0].length == 0) {
      return 0;
    }
    int row = map.length, col = map[0].length;
    map[0][0] = map[0][0] == 1 ? 0 : 1;
    for (int i = 1; i < row; i++) {
      map[i][0] = map[i][0] == 1 ? 0 : map[i - 1][0];
    }
    for (int i = 1; i < col; i++) {
      map[0][i] = map[0][i] == 1 ? 0 : map[0][i - 1];
    }
    for (int i = 1; i < row; i++) {
      for (int j = 1; j < col; j++) {
        map[i][j] = map[i][j] == 1 ? 0 : map[i - 1][j] + map[i][j - 1];
      }
    }
    return map[row-1][col-1];
}
